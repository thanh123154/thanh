#include <iostream>
#include <cstring>

using namespace std;
char* concat(const char* a, const char* b)
{
    int size=strlen(a) + strlen(b);
    char *p = new char[size];


    strcpy(p, a);
    strcat(p, b);

    return p;

}
int main()
{
    const char* a="Hello";
    const char* b="World";
    char* c=concat(a,b);
    cout<<c;
}
