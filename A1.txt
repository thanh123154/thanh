#include <iostream>
#include <cstring>

using namespace std;
char* concat(const char* a, const char* b)
{
    int size=strlen(a) + strlen(b);
    char *result = new char[size];


    strcpy(result, a);
    strcat(result, b);

    return result;

}
int main()
{
    const char* a="Hello";
    const char* b="World";
    char* result=concat(a,b);
    cout<<result;
}
