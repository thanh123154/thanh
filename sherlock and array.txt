#include <bits/stdc++.h>

using namespace std;
int main() {
  int T;
  cin >> T;

  
  for (int i = 0; i < T; i++) 
  {
    int dem = 0;
    int n;
    int a[1000000];
    cin >> n;
    for (int b = 0; b < n; b++) 
    {
      cin >> a[b];

    }  
      for(int b=0;b<n;b++)
      {
            long sum1 = 0;
             long sum2 = 0;
            for (int j = 0; j < b; j++) 
          {
             sum1 += a[j];
        
           }
              for (int k = b+1; k < n; k++) 
            {
              sum2 += a[k];
             }
         if (sum1 == sum2) 
        {
        dem++;
        break;
        }
      }
    
    if (dem == 1)
      cout << "YES"<<endl;
    else
      cout << "NO" <<endl;
   }

}
